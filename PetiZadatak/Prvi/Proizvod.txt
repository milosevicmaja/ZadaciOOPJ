public class Proizvod {
	private String naziv;
	private double cena;

	public Proizvod() {
		this.naziv = "nepoznat";
		this.cena = 0.0;
	}

	public Proizvod(String naziv, double cena) {
		super();
		if (naziv != null && cena > 0) {
			this.naziv = naziv;
			this.cena = cena;
		} else {
                        this.naziv = "nepoznat";
			this.cena = 0.0;
			throw new ProizvodException("Postoji greska (Naziv je null ili cena nije veca od 0)");
		}
	}

	public String getNaziv() {
		return naziv;
	}

	public void setNaziv(String naziv) {
		if (naziv == null) {
			throw new ProizvodException("Postoji greska (Naziv je null)");
		} else {
			this.naziv = naziv;
		}
	}

	public double getCena() {
		return cena;
	}

	public void setCena(double cena) {
		if (cena <= 0) {
			throw new ProizvodException("Postoji greska (Cena nije veca od 0)");
		} else {
			this.cena = cena;
		}
	}

}
