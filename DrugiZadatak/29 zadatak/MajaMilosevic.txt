public class MajaMilosevic {
private String a;
	
	public String getA() {
		return a;
	}

	public void setA(String a) {
		this.a = a;
	}

	public void print() {
		for (int i = a.length() - 1; i >= 0; i--) {
			int j = i;
			while (j != -1) {
				System.out.print(a.charAt(j));
				j--;
			}
			System.out.print("\n");
		}
	}
}
