public class MajaMilosevic {
    private String a;
	
	public String getA() {
		return a;
	}

	public void setA(String a) {
		this.a = a;
	}

	public void print() {
		for (int i = 0; i < a.length(); i++) {
			int j = 0;
			while (j != i + 1) {
				System.out.print(a.charAt(j));
				j++;
			}
			System.out.print("\n");
		}
	}
}
